116  // NOTE: Second order derivatives a 3D matrices, i.e. tensors
117  // We use the numerator convention (see https://en.wikipedia.org/wiki/Matrix_calculus)
118  // X_i,j,k = d(X_i,j)/d x_k

144  virtual Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic> dStateDelta(const StateVector& x_1, const StateVector& x_2, const ArgumentPosition first_or_second)
145  {
146  assert(x_1.size() == x_2.size());
147  assert(first_or_second == ArgumentPosition::ARG0 || first_or_second == ArgumentPosition::ARG1);
148 
149  if (first_or_second == ArgumentPosition::ARG0)
150  return Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic>::Identity(get_num_state_derivative(), get_num_state_derivative());
151  else
152  return -1.0 * Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic>::Identity(get_num_state_derivative(), get_num_state_derivative());
153  }
154 
155  virtual Hessian ddStateDelta(const StateVector& x_1, const StateVector& x_2, const ArgumentPosition first_or_second)
156  {
157  assert(x_1.size() == x_2.size());
158 
159  // In Euclidean spaces, this is zero.
160  Hessian ddStateDelta;
161  ddStateDelta.setConstant(get_num_state_derivative(), Eigen::MatrixXd::Zero(get_num_state_derivative(), get_num_state_derivative()));
162  return ddStateDelta;
163  }
