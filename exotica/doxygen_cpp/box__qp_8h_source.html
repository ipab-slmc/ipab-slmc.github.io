46 inline BoxQPSolution BoxQP(const Eigen::MatrixXd& H, const Eigen::VectorXd& q,
47  const Eigen::VectorXd& b_low, const Eigen::VectorXd& b_high,
48  const Eigen::VectorXd& x_init, const double gamma,
49  const int max_iterations, const double epsilon, const double lambda)
50 {
51  int it = 0;
52  Eigen::VectorXd delta_xf, x = x_init;
53  std::vector<int> clamped_idx, free_idx;
54  Eigen::VectorXd grad = q + H * x_init;
55  Eigen::MatrixXd Hff, Hfc, Hff_inv;
56 
57  Hff_inv = (Eigen::MatrixXd::Identity(H.rows(), H.cols()) * 1e-5 + H).inverse();
58 
59  if (grad.norm() <= epsilon)
60  return {Hff_inv, x_init, free_idx, clamped_idx};
61 
62  while (grad.norm() > epsilon && it < max_iterations)
63  {
64  ++it;
65  grad = q + H * x;
66  clamped_idx.clear();
67  free_idx.clear();
68 
69  for (int i = 0; i < grad.size(); ++i)
70  {
71  if ((x(i) == b_low(i) && grad(i) > 0) || (x(i) == b_high(i) && grad(i) < 0))
72  clamped_idx.push_back(i);
73  else
74  free_idx.push_back(i);
75  }
76 
77  if (free_idx.size() == 0)
78  return {Hff_inv, x, free_idx, clamped_idx};
79 
80  Hff.resize(free_idx.size(), free_idx.size());
81  Hfc.resize(free_idx.size(), clamped_idx.size());
82 
83  if (clamped_idx.size() == 0)
84  Hff = H;
85  else
86  {
87  for (int i = 0; i < free_idx.size(); ++i)
88  for (int j = 0; j < free_idx.size(); ++j)
89  Hff(i, j) = H(free_idx[i], free_idx[j]);
90 
91  for (int i = 0; i < free_idx.size(); ++i)
92  for (int j = 0; j < clamped_idx.size(); ++j)
93  Hfc(i, j) = H(free_idx[i], clamped_idx[j]);
94  }
95 
96  // NOTE: Array indexing not supported in current eigen version
97  Eigen::VectorXd q_free(free_idx.size()), x_free(free_idx.size()), x_clamped(clamped_idx.size());
98  for (int i = 0; i < free_idx.size(); ++i)
99  {
100  q_free(i) = q(free_idx[i]);
101  x_free(i) = x(free_idx[i]);
102  }
103 
104  for (int j = 0; j < clamped_idx.size(); ++j)
105  x_clamped(j) = x(clamped_idx[j]);
106 
107  Hff_inv = (Eigen::MatrixXd::Identity(Hff.rows(), Hff.cols()) * lambda + Hff).inverse();
108 
109  if (clamped_idx.size() == 0)
110  delta_xf = -Hff_inv * (q_free)-x_free;
111  else
112  delta_xf = -Hff_inv * (q_free + Hfc * x_clamped) - x_free;
113 
114  double f_old = (0.5 * x.transpose() * H * x + q.transpose() * x)(0);
115  const Eigen::VectorXd alpha_space = Eigen::VectorXd::LinSpaced(10, 1.0, 0.1);
116 
117  bool armijo_reached = false;
118  Eigen::VectorXd x_new;
119  for (int ai = 0; ai < alpha_space.rows(); ++ai)
120  {
121  int alpha = alpha_space[ai];
122 
123  x_new = x;
124  for (int i = 0; i < free_idx.size(); ++i)
125  x_new(free_idx[i]) = std::max(std::min(
126  x(free_idx[i]) + alpha * delta_xf(i), b_high(i)),
127  b_low(i));
128 
129  double f_new = (0.5 * x_new.transpose() * H * x_new + q.transpose() * x_new)(0);
130  Eigen::VectorXd x_diff = x - x_new;
131 
132  // armijo criterion>
133  double armijo_coef = (f_old - f_new) / (grad.transpose() * x_diff + 1e-5);
134  if (armijo_coef > gamma)
135  {
136  armijo_reached = true;
137  x = x_new;
138  break;
139  }
140  }
141 
142  // break if no step made
143  if (!armijo_reached) break;
144  }
145 
146  return {Hff_inv, x, free_idx, clamped_idx};
147 }
