229  double forwardTimeDistance(const Motion *a, const Motion *b) const
230  {
231  static const Eigen::VectorXd max_vel = si_->getStateSpace()->as<OMPLTimeIndexedRNStateSpace>()->prob_->vel_limits;
232 
233  const OMPLTimeIndexedRNStateSpace::StateType *sa = static_cast<const OMPLTimeIndexedRNStateSpace::StateType *>(a->state);
234  const OMPLTimeIndexedRNStateSpace::StateType *sb = static_cast<const OMPLTimeIndexedRNStateSpace::StateType *>(b->state);
235 
236  double ta, tb;
237  Eigen::VectorXd qa, qb;
238  si_->getStateSpace()->as<OMPLTimeIndexedRNStateSpace>()->OMPLToExoticaState(a->state, qa, ta);
239  si_->getStateSpace()->as<OMPLTimeIndexedRNStateSpace>()->OMPLToExoticaState(b->state, qb, tb);
240 
241  if (tb < ta) return 1e10;
242  Eigen::VectorXd diff = (qb - qa).cwiseAbs();
243  double min_dt = (diff.array() / max_vel.array()).maxCoeff();
244  if (fabs(tb - ta) < min_dt) return 1e10;
245  return si_->distance(a->state, b->state);
246  }

267  bool correctTime(const Motion *a, Motion *b, bool reverse, bool &changed) const
268  {
269  Eigen::VectorXd max_vel = si_->getStateSpace()->as<OMPLTimeIndexedRNStateSpace>()->prob_->vel_limits;
270  double ta, tb;
271  Eigen::VectorXd qa, qb;
272  si_->getStateSpace()->as<OMPLTimeIndexedRNStateSpace>()->OMPLToExoticaState(a->state, qa, ta);
273  si_->getStateSpace()->as<OMPLTimeIndexedRNStateSpace>()->OMPLToExoticaState(b->state, qb, tb);
274  Eigen::VectorXd diff = (qb - qa).cwiseAbs();
275  double min_dt = (diff.array() / max_vel.array()).maxCoeff();
276  if (fabs(tb - ta) < min_dt)
277  {
278  if (reverse_check_) return false;
279  tb = ta + (reverse ? -min_dt : min_dt);
280  changed = true;
281  }
282  else
283  changed = false;
284  if (tb < si_->getStateSpace()->as<OMPLTimeIndexedRNStateSpace>()->prob_->t_start || tb > si_->getStateSpace()->as<OMPLTimeIndexedRNStateSpace>()->prob_->GetGoalTime()) return false;
285  si_->getStateSpace()->as<OMPLTimeIndexedRNStateSpace>()->ExoticaToOMPLState(qb, tb, b->state);
286  return true;
287  }
