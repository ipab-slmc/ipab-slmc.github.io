79 template <typename Initializer>
80 void AbstractDDPSolver<Initializer>::Solve(Eigen::MatrixXd& solution)
81 {
82  if (!prob_) ThrowNamed("Solver has not been initialized!");
83  Timer planning_timer, backward_pass_timer, line_search_timer;
84 
85  const int T = prob_->get_T();
86  const int NU = prob_->get_num_controls();
87  const int NX = prob_->get_num_positions() + prob_->get_num_velocities();
88 
89  prob_->ResetCostEvolution(GetNumberOfMaxIterations() + 1);
90 
91  // initialize Gain matrices
92  K_gains_.assign(T, Eigen::MatrixXd(NU, NX));
93  k_gains_.assign(T, Eigen::VectorXd(NU, 1));
94 
95  // all of the below are not pointers, since we want to copy over
96  // solutions across iterations
97  Eigen::MatrixXd new_U, global_best_U;
98  solution.resize(T, NU);
99 
100  if (debug_) HIGHLIGHT_NAMED("DDPSolver", "Running DDP solver for max " << GetNumberOfMaxIterations() << " iterations");
101 
102  double last_cost = 0, global_best_cost = 0;
103  int last_best_iteration = 0;
104 
105  for (int iteration = 0; iteration < GetNumberOfMaxIterations(); ++iteration)
106  {
107  // Backwards pass computes the gains
108  backward_pass_timer.Reset();
109  BackwardPass();
110  if (debug_) HIGHLIGHT_NAMED("DDPSolver", "Backward pass complete in " << backward_pass_timer.GetDuration());
111 
112  line_search_timer.Reset();
113  // forward pass to compute new control trajectory
114  // TODO: Configure line-search space from xml
115  // TODO (Wolf): What you are doing is a forward line-search when we may try a
116  // backtracking line search for a performance improvement later - this just as an aside.
117  const Eigen::VectorXd alpha_space = Eigen::VectorXd::LinSpaced(10, 0.1, 1.0);
118  double current_cost = 0, best_alpha = 0;
119 
120  Eigen::MatrixXd ref_x = prob_->get_X(),
121  ref_u = prob_->get_U();
122  // perform a linear search to find the best rate
123  for (int ai = 0; ai < alpha_space.rows(); ++ai)
124  {
125  double alpha = alpha_space(ai);
126  double cost = ForwardPass(alpha, ref_x, ref_u);
127 
128  if (ai == 0 || cost < current_cost)
129  {
130  current_cost = cost;
131  new_U = prob_->get_U();
132  best_alpha = alpha;
133  }
134  }
135 
136  if (debug_)
137  {
138  HIGHLIGHT_NAMED("DDPSolver", "Forward pass complete in " << line_search_timer.GetDuration() << " with cost: " << current_cost << " and alpha " << best_alpha);
139  HIGHLIGHT_NAMED("DDPSolver", "Final state: " << prob_->get_X(T - 1).transpose());
140  }
141 
142  // copy solutions for next iteration
143  if (iteration == 0 || global_best_cost > current_cost)
144  {
145  global_best_cost = current_cost;
146  last_best_iteration = iteration;
147  global_best_U = new_U;
148  best_ref_x_ = ref_x;
149  best_ref_u_ = ref_u;
150  }
151 
152  if (iteration - last_best_iteration > base_parameters_.FunctionTolerancePatience)
153  {
154  if (debug_) HIGHLIGHT_NAMED("DDPSolver", "Early stopping criterion reached. Time: " << planning_timer.GetDuration());
155  break;
156  }
157 
158  if (last_cost - current_cost < base_parameters_.FunctionTolerance && last_cost - current_cost > 0)
159  {
160  if (debug_) HIGHLIGHT_NAMED("DDPSolver", "Function tolerance reached. Time: " << planning_timer.GetDuration());
161  break;
162  }
163 
164  if (debug_ && iteration == GetNumberOfMaxIterations() - 1)
165  HIGHLIGHT_NAMED("DDPSolver", "Max iterations reached. Time: " << planning_timer.GetDuration());
166 
167  last_cost = current_cost;
168  for (int t = 0; t < T - 1; ++t)
169  prob_->Update(new_U.col(t), t);
170  prob_->SetCostEvolution(iteration, current_cost);
171  }
172 
173  // store the best solution found over all iterations
174  for (int t = 0; t < T - 1; ++t)
175  {
176  solution.row(t) = global_best_U.col(t).transpose();
177  prob_->Update(global_best_U.col(t), t);
178  }
179 
180  planning_time_ = planning_timer.GetDuration();
181 }

196 template <typename Initializer>
197 double AbstractDDPSolver<Initializer>::ForwardPass(const double alpha, Eigen::MatrixXdRefConst ref_x, Eigen::MatrixXdRefConst ref_u)
198 {
199  double cost = 0;
200  const int T = prob_->get_T();
201  const Eigen::VectorXd control_limits = dynamics_solver_->get_control_limits();
202  const double dt = dynamics_solver_->get_dt();
203 
204  for (int t = 0; t < T - 1; ++t)
205  {
206  Eigen::VectorXd u = ref_u.col(t);
207  // eq. 12
208  Eigen::VectorXd delta_uk = k_gains_[t] + K_gains_[t] * dynamics_solver_->StateDelta(prob_->get_X(t), ref_x.col(t));
209  u.noalias() += alpha * delta_uk;
210  // clamp controls
211  u = u.cwiseMax(-control_limits).cwiseMin(control_limits);
212 
213  prob_->Update(u, t);
214  cost += dt * (prob_->GetControlCost(t) + prob_->GetStateCost(t));
215  }
216 
217  // add terminal cost
218  cost += prob_->GetStateCost(T - 1);
219  return cost;
220 }

222 template <typename Initializer>
223 Eigen::VectorXd AbstractDDPSolver<Initializer>::GetFeedbackControl(Eigen::VectorXdRefConst x, int t) const
224 {
225  const Eigen::VectorXd control_limits = dynamics_solver_->get_control_limits();
226  Eigen::VectorXd delta_uk = k_gains_[t] + K_gains_[t] * dynamics_solver_->StateDelta(x, best_ref_x_.col(t));
227 
228  Eigen::VectorXd u = best_ref_u_.col(t) + delta_uk;
229  return u.cwiseMax(-control_limits).cwiseMin(control_limits);
230 }
